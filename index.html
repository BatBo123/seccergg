// ============================= // OpenAI API Key // ============================= const OPENAI_API_KEY = "여기에_새_API_KEY_입력"; function runAutoDataGenerator(e) { if (!e || !e.range) return; const sheet = e.source.getActiveSheet(); const range = e.range; if (range.getColumn() != 1) return; if (range.getRow() == 1) return; const word = range.getValue().toString().trim(); if (word == "") { sheet.getRange(range.getRow(),2,1,3).clearContent(); return; } sheet.getRange(range.getRow(),2).setValue("설명 생성 중..."); sheet.getRange(range.getRow(),3).setValue("카테고리 생성 중..."); sheet.getRange(range.getRow(),4).setValue("태그 생성 중..."); SpreadsheetApp.flush(); try{ const result = callChatGPT(word); sheet.getRange(range.getRow(),2).setValue(result.description); sheet.getRange(range.getRow(),3).setValue(result.category); sheet.getRange(range.getRow(),4).setValue(result.tags); }catch(err){ sheet.getRange(range.getRow(),2).setValue(err.message); sheet.getRange(range.getRow(),3).clearContent(); sheet.getRange(range.getRow(),4).clearContent(); } } // ====================================== // ChatGPT 호출 // ====================================== function callChatGPT(word){ const url = "https://api.openai.com/v1/chat/completions"; const prompt = ` 당신은 체육 및 스포츠 데이터 분석 전문가입니다. 사용자가 입력한 단어를 분석하여 아래 JSON만 출력하세요. 입력 단어 ${word} 설명 규칙 - 1~2문장 - 체육 전공자가 이해하기 쉽게 작성 카테고리 다음 중 하나만 선택 배구 축구 농구 야구 육상 웨이트 체력운동 스트레칭 운동생리학 운동역학 스포츠심리 재활운동 스포츠의학 체육교육 스포츠용어 기타 태그 - 5~8개 - 쉼표(,)로 구분 반드시 아래 JSON만 출력 { "description":"", "category":"", "tags":"" } `; const payload = { model:"gpt-4o-mini", messages:[ { role:"system", content:"너는 체육 데이터 생성 AI이다. 반드시 JSON만 출력한다." }, { role:"user", content:prompt } ], response_format:{ type:"json_object" }, temperature:0.4 }; const options = { method:"post", contentType:"application/json", headers:{ Authorization:"Bearer " + OPENAI_API_KEY }, payload:JSON.stringify(payload), muteHttpExceptions:true }; const response = UrlFetchApp.fetch(url,options); const code = response.getResponseCode(); const body = response.getContentText(); if(code != 200){ throw new Error(body); } const json = JSON.parse(body); return JSON.parse(json.choices[0].message.content); }
